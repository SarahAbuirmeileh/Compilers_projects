# Original Grammar

Program -> Header Declarations Block.
Header -> program id(input,output) ;
Declarations ->VAR VariableDeclarations | <epsilon>
VariableDeclarations -> VariableDeclarations VariableDeclaration
                    | VariableDeclaration

VariableDeclaration -> IdentifierList : Type ;
IdentifierList -> IdentifierList , id | id
Type -> integer | real
Block -> begin Statements end

Statements -> Statements ; Statement | Statement
Statement -> id := Expression
            | Block
            | if Expression then Statement else Statement
            | repeat Statement until Expression
            | writeln(SimpleExpression)
            | <epsilon>

ExpressionList -> ExpressionList , Expression | Expression
Expression -> SimpleExpression relop SimpleExpression | SimpleExpression
SimpleExpression -> Term| addop Term| SimpleExpression addop Term
Term -> Term mulop Factor| Factor
Factor -> id | num | ( Expression ) | not Factor

# Grammar After Removing Ambiguity 

Program -> Header Declarations Block.
Header -> program id (input, output) ;
Declarations -> VAR VariableDeclarations | <epsilon>
VariableDeclarations -> VariableDeclarations VariableDeclaration | VariableDeclaration

VariableDeclaration -> IdentifierList : Type ;
IdentifierList -> IdentifierList , id | id
Type -> integer | real

Block -> begin Statements end
Statements -> Statements ; Statement | Statement

Statement -> id := Expression
            | Block
            | IfStatement
            | repeat Statements until Expression
            | writeln(SimpleExpression)

IfStatement  -> Matched | Unmatched
Matched   -> if Expression then Matched else Matched
Unmatched -> if Expression then IfStatement 
           | if Expression then Matched else Unmatched

ExpressionList -> ExpressionList , Expression | Expression
Expression -> SimpleExpression relop SimpleExpression  | SimpleExpression
SimpleExpression -> SimpleExpression addop Term | addop Term | Term
Term -> Term mulop Factor | Factor
Factor -> id | num | ( Expression ) | not Factor

# Grammar After Removing Left Recursion

Program -> Header Declarations Block.
Header -> program id (input, output) ;
Declarations -> VAR VariableDeclarations | <epsilon>
VariableDeclarations -> VariableDeclaration R1
R1 -> VariableDeclaration R1 | <epsilon>

VariableDeclaration -> IdentifierList : Type ;
IdentifierList -> id MoreIdList
MoreIdList -> , id OtherIdIist | <epsilon>
Type -> integer | real

Block -> begin Statements end
Statements -> Statement R3 
R3 -> ; Statement R3 | <epsilon>

Statement -> id := Expression
            | Block
            | IfStatement
            | repeat Statements until Expression
            | writeln(SimpleExpression)

IfStatement  -> Matched | Unmatched
Matched   -> if Expression then Matched else Matched
Unmatched -> if Expression then IfStatement 
           | if Expression then Matched else Unmatched

ExpressionList -> Expression R4
R4 -> , Expression R4 | <epsilon>
Expression -> SimpleExpression relop SimpleExpression  | SimpleExpression

TrivialCase -> addop Term | Term
SimpleExpression -> TrivialCase R5
R5 -> addop Term R5 | <epsilon>

Term -> Factor R6
R6 -> mulop Factor R6 | <epsilon>
Factor -> id | num | ( Expression ) | not Factor

# Grammar After Left Factoring

Program -> Header Declarations Block.
Header -> program id (input, output) ;
Declarations -> VAR VariableDeclarations | <epsilon>
VariableDeclarations -> VariableDeclaration MoreVariableDeclarations
MoreVariableDeclarations -> VariableDeclaration MoreVariableDeclarations | <epsilon>

VariableDeclaration -> IdentifierList : Type ;
IdentifierList -> id OtherIdIist
MoreIdList -> , id MoreIdList | <epsilon>
Type -> integer | real

Block -> begin Statements end
Statements -> Statement R3 
R3 -> ; Statement R3 | <epsilon>

Statement -> id := Expression
                  | Block
                  | IfStatement
                  | repeat Statements until Expression
                  | writeln(SimpleExpression)

IfStatement -> if Expression then Matched OptionalTail 
Matched -> if Expression then Matched else Matched 
OptionalTail -> else Tail | <epsilon>
Tail -> if Expression then Tail 

ExpressionList -> Expression R4
R4 -> , Expression R4 | <epsilon>
Expression -> SimpleExpression A1
A1 -> relop SimpleExpression  | <epsilon>

TrivialCase -> addop Term | Term
SimpleExpression -> TrivialCase  R5
R5 -> addop Term R5 | <epsilon>

Term -> Factor R6
R6 -> mulop Factor R6 | <epsilon>
Factor -> id | num | ( Expression ) | not Factor
